def colorGradientHLS(imIn, se = Morpho.getDefaultSE(), convertFirstToHLS = True):
  imOut = Image(imIn, imtype = 'UINT8')
  r = gradientHLS(imIn, imOut, se, convertFirstToHLS)
  if r == 1:
    return imOut
  return None
