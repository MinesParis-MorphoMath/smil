
SET(LIB_NAME smilCoreGui)

FILE(GLOB ${LIB_NAME}_UIS Qt/*.ui)
FILE(GLOB ${LIB_NAME}_SRCS Qt/*.cpp)
FILE(GLOB ${LIB_NAME}_QT_HEADERS Qt/*.h)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

QT4_WRAP_UI(${LIB_NAME}_UIS_H ${${LIB_NAME}_UIS})
QT4_WRAP_CPP(${LIB_NAME}_MOC ${${LIB_NAME}_QT_HEADERS})


FILE(GLOB SRCS Qt/*.cpp)
ADD_LIBRARY(smilCoreGui ${SRCS} ${${LIB_NAME}_UIS_H} ${${LIB_NAME}_MOC})
TARGET_LINK_LIBRARIES(smilCoreGui ${QT_LIBRARIES})
